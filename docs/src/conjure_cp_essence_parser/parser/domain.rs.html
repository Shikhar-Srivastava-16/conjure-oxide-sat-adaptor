6/// Parse an Essence variable domain into its Conjure AST representation.
7pub fn parse_domain(domain: Node, source_code: &str) -> Result<Domain, EssenceParseError> {
8    match domain.kind() {
9        "domain" => parse_domain(domain.child(0).expect("No domain found"), source_code),
10        "bool_domain" => Ok(Domain::Bool),
11        "int_domain" => Ok(parse_int_domain(domain, source_code)),
12        "identifier" => {
13            let variable_name = &source_code[domain.start_byte()..domain.end_byte()];
14            Ok(Domain::Reference(Name::user(variable_name)))
15        }
16        "tuple_domain" => parse_tuple_domain(domain, source_code),
17        "matrix_domain" => parse_matrix_domain(domain, source_code),
18        "record_domain" => parse_record_domain(domain, source_code),
19        _ => panic!("{} is not a supported domain type", domain.kind()),
20    }
21}

23/// Parse an integer domain. Can be a single integer or a range.
24fn parse_int_domain(int_domain: Node, source_code: &str) -> Domain {
25    if int_domain.child_count() == 1 {
26        Domain::Int(vec![Range::Bounded(i32::MIN, i32::MAX)])
27    } else {
28        let mut ranges: Vec<Range<i32>> = Vec::new();
29        let range_list = int_domain
30            .child_by_field_name("ranges")
31            .expect("No range list found (expression ranges not supported yet");
32        for int_range in named_children(&range_list) {
33            match int_range.kind() {
34                "integer" => {
35                    let integer_value = &source_code[int_range.start_byte()..int_range.end_byte()]
36                        .parse::<i32>()
37                        .unwrap();
38                    ranges.push(Range::Single(*integer_value));
39                }
40                "int_range" => {
41                    let lower_bound: Option<i32>;
42                    let upper_bound: Option<i32>;
43                    if let Some(lower_bound_node) = int_range.child_by_field_name("lower") {
44                        lower_bound = Some(
45                            source_code[lower_bound_node.start_byte()..lower_bound_node.end_byte()]
46                                .parse::<i32>()
47                                .unwrap(),
48                        );
49                    } else {
50                        lower_bound = None;
51                    }
52                    if let Some(upper_bound_node) = int_range.child_by_field_name("upper") {
53                        upper_bound = Some(
54                            source_code[upper_bound_node.start_byte()..upper_bound_node.end_byte()]
55                                .parse::<i32>()
56                                .unwrap(),
57                        );
58                    } else {
59                        upper_bound = None;
60                    }
61
62                    match (lower_bound, upper_bound) {
63                        (Some(lb), Some(ub)) => ranges.push(Range::Bounded(lb, ub)),
64                        (Some(lb), None) => ranges.push(Range::Bounded(lb, i32::MAX)),
65                        (None, Some(ub)) => ranges.push(Range::Bounded(i32::MIN, ub)),
66                        _ => panic!("Unsupported int range type"),
67                    }
68                }
69                _ => panic!("unsupported int range type"),
70            }
71        }
72        Domain::Int(ranges)
73    }
74}

76fn parse_tuple_domain(tuple_domain: Node, source_code: &str) -> Result<Domain, EssenceParseError> {
77    let mut domains: Vec<Domain> = Vec::new();
78    for domain in named_children(&tuple_domain) {
79        domains.push(parse_domain(domain, source_code)?);
80    }
81    Ok(Domain::Tuple(domains))
82}

84fn parse_matrix_domain(
85    matrix_domain: Node,
86    source_code: &str,
87) -> Result<Domain, EssenceParseError> {
88    let mut domains: Vec<Domain> = Vec::new();
89    let index_domain_list = matrix_domain
90        .child_by_field_name("index_domain_list")
91        .expect("No index domains found for matrix domain");
92    for domain in named_children(&index_domain_list) {
93        domains.push(parse_domain(domain, source_code)?);
94    }
95    let value_domain = parse_domain(
96        matrix_domain.child_by_field_name("value_domain").ok_or(
97            EssenceParseError::syntax_error(
98                "Expected a value domain".to_string(),
99                Some(matrix_domain.range()),
100            ),
101        )?,
102        source_code,
103    )?;
104    Ok(Domain::Matrix(Box::new(value_domain), domains))
105}

107fn parse_record_domain(
108    record_domain: Node,
109    source_code: &str,
110) -> Result<Domain, EssenceParseError> {
111    let mut record_entries: Vec<RecordEntry> = Vec::new();
112    for record_entry in named_children(&record_domain) {
113        let name_node = record_entry
114            .child_by_field_name("name")
115            .expect("No name found for record entry");
116        let name = Name::user(&source_code[name_node.start_byte()..name_node.end_byte()]);
117        let domain_node = record_entry
118            .child_by_field_name("domain")
119            .expect("No domain found for record entry");
120        let domain = parse_domain(domain_node, source_code)?;
121        record_entries.push(RecordEntry { name, domain });
122    }
123    Ok(Domain::Record(record_entries))
124}
